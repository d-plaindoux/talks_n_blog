Constructeur et cas limite

class Personne {
 let nom : String
 var age : Int

 init(nom:String, age:Int) {
  self.nom = nom
  self.age = age
 } 

 convenience init?(nom:String, année:Int) {
  if annéeCourante() - année < 0 { return nil }
  self.init(nom:nom, age:annéeCourante() - année)
 }
}
